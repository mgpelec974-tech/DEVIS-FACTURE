Google Sheets (Grand Livre)

📌 Configuration Google Sheets : Pour synchroniser vos données, suivez ces étapes :

1. Créez un Google Sheet et notez son ID (dans l'URL entre /d/ et /edit)

2. Créez un projet Apps Script (Extensions → Apps Script) avec un webhook POST

3. Publiez le script comme application web et copiez l'URL ici

Code Apps Script à copier :

function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  const ss = SpreadsheetApp.openById(data.sheetId);
  
  if (data.type === 'facture') {
    const sh = ss.getSheetByName('Grand Livre') || ss.insertSheet('Grand Livre');
    if (sh.getLastRow() === 0) {
      sh.appendRow(['N°','Date','Client','Description','HT','TVA','TTC','Statut','Mode Paiement']);
    }
    // Mettre à jour ou ajouter
    sh.appendRow([data.num, data.date, data.client, data.titre,
      data.ht, data.tva, data.ttc, data.statut, data.modePaiement]);
  }
  
  if (data.type === 'materiel') {
    const sh = ss.getSheetByName('Matériaux') || ss.insertSheet('Matériaux');
    if (sh.getLastRow() === 0) {
      sh.appendRow(['Désignation','Quantité','Prix unitaire','Total']);
    }
    data.lignes.forEach(l => sh.appendRow([l.desc, l.qte, l.pu, l.total]));
  }
  
  return ContentService.createTextOutput('OK');
}
